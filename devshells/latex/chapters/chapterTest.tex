#include <iostream>
#include <vector>

int main() {
    std::vector<std::string> msg = {"Hello", "Minted"};
    for (const auto& word : msg) {
        std::cout << word << " ";
    }
    return 0;
}
\end{minted}

\subsection{Custom 'code' Environment}
% \begin{noindent}
\begin{code}{python}
def calculate_area(radius):
import math
if radius < 0:
raise ValueError("Radius cannot be negative")
return math.pi * (radius ** 2)

print(f"Area: {calculate_area(5):.2f}")
\end{code}
% \end{noindent}
